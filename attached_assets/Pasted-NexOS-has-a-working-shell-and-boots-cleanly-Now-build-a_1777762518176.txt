1D — Create kernel/drivers/fb.c
Implement all functions from fb.h:

  framebuffer_t fb = {0};

  void fb_init(uint64_t addr, uint32_t w, uint32_t h,
               uint32_t pitch, uint8_t bpp) {
      fb.addr  = (uint32_t*)addr;
      fb.width = w; fb.height = h;
      fb.pitch = pitch; fb.bpp = bpp;
      fb.initialized = 1;
  }

  void fb_put_pixel(int x, int y, uint32_t color) {
      if (x<0||y<0||x>=(int)fb.width||y>=(int)fb.height) 
          return;
      uint32_t *row = (uint32_t*)
          ((uint8_t*)fb.addr + y * fb.pitch);
      row[x] = color;
  }

  void fb_fill_rect(int x, int y, int w, int h, uint32_t c) {
      for (int row=y; row<y+h; row++)
          for (int col=x; col<x+w; col++)
              fb_put_pixel(col, row, c);
  }

  void fb_fill_rounded_rect(int x,int y,int w,int h,
                            int r,uint32_t c):
    Use midpoint circle algorithm for corners.
    Fill center rect + 4 corner quadrants.

  uint32_t fb_blend(uint32_t fg, uint32_t bg, uint8_t alpha):
    uint8_t r = ((fg>>16&0xFF)*alpha + 
                 (bg>>16&0xFF)*(255-alpha)) / 255;
    uint8_t g = ((fg>>8&0xFF)*alpha + 
                 (bg>>8&0xFF)*(255-alpha)) / 255;
    uint8_t b = ((fg&0xFF)*alpha + 
                 (bg&0xFF)*(255-alpha)) / 255;
    return (r<<16)|(g<<8)|b;

  void fb_scroll_up(int pixels, uint32_t bg):
    memmove(fb.addr,
            (uint8_t*)fb.addr + pixels * fb.pitch,
            (fb.height - pixels) * fb.pitch);
    fb_fill_rect(0, fb.height-pixels, fb.width, pixels, bg);

  fb_draw_line: Bresenham's line algorithm
  fb_draw_circle: Midpoint circle algorithm (outline only)
  fb_fill_circle: Fill using horizontal spans

2A — Initialization
  static int mouse_x, mouse_y;
  static uint8_t mouse_btns;
  static uint8_t mouse_cycle = 0;
  static uint8_t mouse_bytes[3];

  void mouse_wait_write(void) {
      int t = 100000;
      while (t-- && (inb(0x64)&2));
  }
  void mouse_wait_read(void) {
      int t = 100000;
      while (t-- && !(inb(0x64)&1));
  }
  void mouse_write(uint8_t val) {
      mouse_wait_write();
      outb(0x64, 0xD4);
      mouse_wait_write();
      outb(0x60, val);
  }
  uint8_t mouse_read(void) {
      mouse_wait_read();
      return inb(0x60);
  }

  void mouse_init():
    // Enable aux device
    mouse_wait_write(); outb(0x64, 0xA8);
    // Enable aux interrupts
    mouse_wait_write(); outb(0x64, 0x20);
    mouse_wait_read();
    uint8_t status = inb(0x60) | 2;
    mouse_wait_write(); outb(0x64, 0x60);
    mouse_wait_write(); outb(0x60, status);
    // Set defaults + enable
    mouse_write(0xF6); mouse_read(); // ack
    mouse_write(0xF4); mouse_read(); // ack
    // Install IRQ12
    idt_register_irq(12, mouse_irq_handler);
    mouse_x = fb.width / 2;
    mouse_y = fb.height / 2;
    klog(INFO, "Mouse: PS/2 initialized");

2B — IRQ12 handler + packet decoding
  void mouse_irq_handler(registers_t *r):
    mouse_bytes[mouse_cycle] = inb(0x60);
    if (mouse_cycle == 0 && !(mouse_bytes[0] & 0x08))
        return; // bit 3 must always be set in byte 0
    mouse_cycle++;
    if (mouse_cycle == 3) {
        mouse_cycle = 0;
        mouse_btns = mouse_bytes[0] & 0x07;
        int dx = (int8_t)mouse_bytes[1];
        int dy = -(int8_t)mouse_bytes[2];
        // Handle overflow bits
        if (mouse_bytes[0] & 0x40) dx = (dx>0)?-127:127;
        if (mouse_bytes[0] & 0x80) dy = (dy>0)?-127:127;
        mouse_x += dx;
        mouse_y += dy;
        if (mouse_x < 0) mouse_x = 0;
        if (mouse_y < 0) mouse_y = 0;
        if (mouse_x >= (int)fb.width)  
            mouse_x = fb.width-1;
        if (mouse_y >= (int)fb.height) 
            mouse_y = fb.height-1;
        mouse_needs_redraw = 1;
    }

  #define TERM_COLS  90
  #define TERM_ROWS  28
  #define TERM_PAD    8

  typedef struct {
      window_t *win;
      char buf[TERM_ROWS][TERM_COLS+1];
      int col, row;
      char input[256];
      int input_len;
      uint32_t fg, bg;

  } term_app_t;

  term_app_t *term_create(int x, int y):
    window_t *win = wm_new(x, y, 
        TERM_COLS*8 + TERM_PAD*2,
        TERM_ROWS*16 + WM_TITLEBAR_H + TERM_PAD*2,
        "Terminal");
    win->on_paint = term_paint;
    win->on_key   = term_key;
    win->on_close = term_close;
    // Allocate term state
    term_app_t *t = kmalloc(sizeof(term_app_t));
    memset(t, 0, sizeof(term_app_t));
    t->win = win;
    t->fg  = COL_TEXT;
    t->bg  = COL_MANTLE;
    win->userdata = t;
    // Print welcome + prompt
    term_puts(t, "NexOS Terminal\n");
    term_puts(t, "Type 'help' for commands\n\n");
    term_prompt(t);
    return t;

  term_paint(window_t *win):
    term_app_t *t = win->userdata;
    // Fill client area
    fb_fill_rect(win->x, win->y+WM_TITLEBAR_H,
        win->w, win->h-WM_TITLEBAR_H, t->bg);
    // Draw each line of buffer
    for (int r=0; r<TERM_ROWS; r++)
        font_puts(win->x+TERM_PAD, 
                  win->y+WM_TITLEBAR_H+TERM_PAD+r*16,
                  t->buf[r], t->fg, t->bg);
    // Draw cursor (blinking block)
    uint64_t tick = timer_get_ticks();
    if ((tick/500) % 2 == 0) {  // blink every 500ms
        int cx = win->x + TERM_PAD + t->col*8;
        int cy = win->y + WM_TITLEBAR_H + TERM_PAD + t->row*16;
        fb_fill_rect(cx, cy, 8, 16, COL_BLUE);
    }

  term_key(window_t *win, char key):
    term_app_t *t = win->userdata;
    if (key == '\n') {
        term_newline(t);
        // Execute command via nsh command dispatch
        nsh_exec_command(t->input, t);
        t->input_len = 0;
        memset(t->input, 0, 256);
        term_prompt(t);
    } else if (key == '\b') {
        if (t->input_len > 0) {
            t->input[--t->input_len] = 0;
            // Remove char from display
            t->col--;
            if (t->col < 0) t->col = 0;
            t->buf[t->row][t->col] = ' ';
        }
    } else {
        t->input[t->input_len++] = key;
        t->buf[t->row][t->col++] = key;
        if (t->col >= TERM_COLS) {
            t->col = 0;
            term_scroll_if_needed(t);
        }
    }
    wm_invalidate(win);

  term_puts(term_app_t *t, const char *s):
    while (*s) {
        if (*s == '\n') {
            t->col = 0;
            t->row++;
            term_scroll_if_needed(t);
        } else {
            t->buf[t->row][t->col++] = *s;
        }
        s++;
    }

  void gui_main(void):
    // 1. Init subsystems
    mouse_init();
    wm_init();
    taskbar_init();

    // 2. Draw initial desktop
    desktop_draw();
    taskbar_draw();

    // 3. Open startup windows
    term_create(60, 60);      // Terminal at top-left area
    files_create(560, 80);    // Files on the right
    // sysinfo opens on first launch, then user can close

    // 4. Main event loop (target 30fps)
    uint64_t last_frame = 0;
    uint64_t last_clock = 0;
    uint64_t last_blink = 0;
    int prev_left = 0, prev_right = 0;
    int prev_mx = -1, prev_my = -1;

    while (1) {
        uint64_t now = timer_get_ticks();

        // Keyboard events
        while (keyboard_available()) {
            char key = keyboard_getchar();
            // Global shortcuts
            if (key == 3) {  // Ctrl+T (ASCII ETX)
                term_create(80 + (rand()%200), 
                            80 + (rand()%100));
            } else {
                wm_handle_key(key);
                if (launcher_visible)
                    launcher_handle_key(key);
            }
        }

        // Mouse events
        int mx = mouse_get_x();
        int my = mouse_get_y();
        int left = mouse_left();
        int right = mouse_right();

        int left_click  = left && !prev_left;
        int right_click = right && !prev_right;
        int released    = !left && prev_left;

        if (left_click || right_click || 
            mx!=prev_mx || my!=prev_my) {
            // Check launcher first
            if (launcher_visible && left_click) {
                launcher_handle_click(mx, my);
            } else if (right_click) {
                // Right click on desktop area?
                if (my < TB_Y)
                    launcher_show(mx, my);
            } else {
                wm_handle_mouse(mx, my, left, right);
                // Taskbar click
                if (my >= TB_Y && left_click)
                    taskbar_handle_click(mx, my);
            }
            prev_mx=mx; prev_my=my;
        }
        if (released) wm_handle_mouse_release(mx, my);
        prev_left=left; prev_right=right;

        // Frame render at ~30fps
        if (now - last_frame >= 33) {
            last_frame = now;
            desktop_draw();
            wm_render_all();
            taskbar_draw();
            launcher_draw();
            cursor_restore();
            cursor_draw(mx, my);
        }

        // Clock update every second
        if (now - last_clock >= 1000) {
            last_clock = now;
            taskbar_update();
        }

        // Yield CPU to avoid busy-loop burning 100%
        asm volatile("hlt");
    }
